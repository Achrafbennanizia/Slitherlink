121 int calculateOptimalParallelDepth()
122 {
123 int totalCells = grid.n * grid.m;
124 int clueCount = count_if(grid.clues.begin(), grid.clues.end(), [](int c)
125 { return c >= 0; });
126 double density = (double)clueCount / totalCells;

128 int depth = (totalCells <= 25) ? 8 : (totalCells <= 49) ? 12

140 void buildEdges()
141 {

149 clueCells.clear();

152 auto pointId = [m](int r, int c)
153 { return r * (m + 1) + c; };
154 int idx = 0;
155
156 // Build horizontal edges
157 for (int r = 0; r <= n; ++r)
158 for (int c = 0; c < m; ++c)
159 {
160 Edge e{pointId(r, c), pointId(r, c + 1),
161 (r > 0) ? grid.cellIndex(r - 1, c) : -1,
162 (r < n) ? grid.cellIndex(r, c) : -1};
163 edges.push_back(e);
164 horizEdgeIndex[r * m + c] = idx;
165 if (e.cellA >= 0)
166 cellEdges[e.cellA].push_back(idx);
167 if (e.cellB >= 0)
168 cellEdges[e.cellB].push_back(idx);
169 pointEdges[e.u].push_back(idx);
170 pointEdges[e.v].push_back(idx);
171 idx++;
172 }
173
174 // Build vertical edges
175 for (int r = 0; r < n; ++r)
176 for (int c = 0; c <= m; ++c)
177 {
178 Edge e{pointId(r, c), pointId(r + 1, c),
179 (c > 0) ? grid.cellIndex(r, c - 1) : -1,
180 (c < m) ? grid.cellIndex(r, c) : -1};
181 edges.push_back(e);
182 vertEdgeIndex[r * (m + 1) + c] = idx;
183 if (e.cellA >= 0)
184 cellEdges[e.cellA].push_back(idx);
185 if (e.cellB >= 0)
186 cellEdges[e.cellB].push_back(idx);
187 pointEdges[e.u].push_back(idx);
188 pointEdges[e.v].push_back(idx);
189 idx++;
190 }

192 for (size_t i = 0; i < grid.clues.size(); ++i)
193 if (grid.clues[i] >= 0)
194 clueCells.push_back((int)i);

214 bool applyDecision(State &s, int edgeIdx, int val) const
215 {
216 if (s.edgeState[edgeIdx] == val)
217 return true;
218 if (s.edgeState[edgeIdx] != 0)
219 return false;
220
221 s.edgeState[edgeIdx] = (char)val;
222 const Edge &e = edges[edgeIdx];
223
224 s.pointUndecided[e.u]--;
225 s.pointUndecided[e.v]--;
226 if (e.cellA >= 0)
227 s.cellUndecided[e.cellA]--;
228 if (e.cellB >= 0)
229 s.cellUndecided[e.cellB]--;
230
231 if (val == 1)
232 {
233 int du = ++s.pointDegree[e.u];
234 int dv = ++s.pointDegree[e.v];
235 if (du > 2 || dv > 2)
236 return false;
237
238 if (e.cellA >= 0)
239 {
240 int cnt = ++s.cellEdgeCount[e.cellA];
241 if (grid.clues[e.cellA] >= 0 && cnt > grid.clues[e.cellA])
242 return false;
243 }
244 if (e.cellB >= 0)
245 {
246 int cnt = ++s.cellEdgeCount[e.cellB];
247 if (grid.clues[e.cellB] >= 0 && cnt > grid.clues[e.cellB])
248 return false;
249 }
250 }

498 int selectNextEdge(const State &s) const
499 {
500 int bestEdge = -1, bestScore = -1000;
501
502 auto scoreCell = [&](int cellIdx) -> int
503 {
504 if (cellIdx < 0 || grid.clues[cellIdx] < 0)
505 return 0;
506 int clue = grid.clues[cellIdx], cnt = s.cellEdgeCount[cellIdx], und = s.cellUndecided[cellIdx];
507 if (und == 0)
508 return 0;
509 int need = clue - cnt;
510 return (need == und || need == 0) ? 2000 : (und == 1) ? 1500
511 : (und <= 2) ? 1000
512 : max(0, 100 - abs(need * 2 - und));
513 };
514
515 for (int i = 0; i < (int)edges.size(); ++i)
516 {
517 if (s.edgeState[i] != 0)
518 continue;
519 const Edge &e = edges[i];
520 int degU = s.pointDegree[e.u], degV = s.pointDegree[e.v];
521 int undU = s.pointUndecided[e.u], undV = s.pointUndecided[e.v];
522
523 int score = ((degU == 1 || degV == 1) ? 10000 : 0) +
524 ((degU == 0 && undU == 2) || (degV == 0 && undV == 2) ? 5000 : 0) +
525 scoreCell(e.cellA) + scoreCell(e.cellB);
526
527 if (score > bestScore)
528 {
529 bestScore = score;
530 bestEdge = i;
531 if (bestScore >= 10000)
532 return bestEdge;
533 }
534 }
535 return bestEdge >= 0 ? bestEdge : (int)edges.size();
536 }

736 void search(State s, int depth)
737 {
738 if (!findAll && stopAfterFirst.load(memory_order_relaxed))
739 return;
740
741 if (!quickValidityCheck(s))
742 return;
743
744 if (!propagateConstraints(s))
745 return;
746
747 int edgeIdx = selectNextEdge(s);
748 if (edgeIdx == (int)edges.size())
749 {
750 finalCheckAndStore(s);
751 return;
752 }
753
754 const Edge &edge = edges[edgeIdx];
755 bool canOff = true;
756 bool canOn = true;
757
758 int degU = s.pointDegree[edge.u];
759 int degV = s.pointDegree[edge.v];
760 int undU = s.pointUndecided[edge.u];
761 int undV = s.pointUndecided[edge.v];
762
763 if ((degU == 1 && undU == 1) || (degV == 1 && undV == 1))
764 canOff = false;
765 if (degU >= 2 || degV >= 2)
766 canOn = false;
767
768 State offState;
769 if (canOff)
770 {
771 offState = s;
772 if (!(applyDecision(offState, edgeIdx, -1) && quickValidityCheck(offState) && propagateConstraints(offState)))
773 canOff = false;
774 }
775
776 State onState;
777 if (canOn)
778 {
779 onState = s;
780 if (!(applyDecision(onState, edgeIdx, 1) && quickValidityCheck(onState) && propagateConstraints(onState)))
781 canOn = false;
782 }
783
784 if (!canOn && !canOff)
785 return;
786 if (canOn && !canOff)
787 {
788 search(std::move(onState), depth + 1);
789 return;
790 }
791 if (!canOn && canOff)
792 {
793 search(std::move(offState), depth + 1);
794 return;
795 }
796
797#ifdef USE_TBB
798 if (depth < maxParallelDepth)
799 {
800 tbb::task_group g;
801 g.run([this, off = std::move(offState), depth]()
802 {
803 State local = off;
804 search(std::move(local), depth + 1); });
805 search(std::move(onState), depth + 1);
806 g.wait();
807 }
808 else
809 {
810 search(std::move(offState), depth + 1);
811 if (!findAll && stopAfterFirst.load(memory_order_relaxed))
812 return;
813 search(std::move(onState), depth + 1);
814 }
815#else
816 bool doParallel = (depth < maxParallelDepth &&
817 activeThreads.load(memory_order_relaxed) < maxThreads);
818 if (doParallel)
819 {
820 activeThreads.fetch_add(1, memory_order_relaxed);
821 auto fut = std::async(std::launch::async, [this, off = std::move(offState), depth]() mutable
822 {
823 search(std::move(off), depth + 1);
824 activeThreads.fetch_sub(1, memory_order_relaxed); });
825 search(std::move(onState), depth + 1);
826 fut.get();
827 }
828 else
829 {
830 search(std::move(offState), depth + 1);
831 if (!findAll && stopAfterFirst.load(memory_order_relaxed))
832 return;
833 search(std::move(onState), depth + 1);
834 }
835#endif
836 }

893 auto isHorizOn = [&](int r, int c) -> bool

895 int idx = horizEdgeIndex[r * m + c];

897 };

899 {
900 int idx = vertEdgeIndex[r * (m + 1) + c];
901 return sol.edgeState[idx] == 1;
902 };
903
904 for (int r = 0; r <= n; ++r)
905 {
906 string line;
907 for (int c = 0; c < m; ++c)
908 {
909 line += "+";
910 line += (isHorizOn(r, c) ? "-" : " ");
911 }
912 line += "+";
913 cout << line << "\n";
914
915 if (r == n)
916 break;
917
918 string vline;
919 for (int c = 0; c < m; ++c)
920 {
921 vline += (isVertOn(r, c) ? "|" : " ");
922 int clue = grid.clues[grid.cellIndex(r, c)];
923 char ch = ' ';
924 if (clue >= 0)
925 ch = char('0' + clue);
926 vline += ch;
927 }
928 vline += (isVertOn(r, m) ? "|" : " ");
929 cout << vline << "\n";
930 }
931
932 cout << "Cycle (point coordinates row,col):\n";
933 for (size_t i = 0; i < sol.cyclePoints.size(); ++i)
934 {
935 auto [r, c] = sol.cyclePoints[i];
936 cout << "(" << r << "," << c << ")";
937 if (i + 1 < sol.cyclePoints.size())
938 cout << " -> ";

955Grid readGridFromFile(const string &filename)
956{
957 ifstream in(filename);
958 if (!in)
959 {
960 throw runtime_error("Could not open file " + filename);
961 }
962 Grid g;
963 in >> g.n >> g.m;
964 string line;
965 getline(in, line);
966
967 g.clues.assign(g.n * g.m, -1);
968
969 for (int r = 0; r < g.n; ++r)
970 {
971 if (!getline(in, line))
972 {
973 throw runtime_error("Not enough grid lines in file");
974 }
975 if (line.empty())
976 {
977 r--;
978 continue;
979 }
980 vector<int> row;
981 for (char ch : line)
982 {
983 if (ch == ' ' || ch == '\t')
984 continue;
985 if (ch >= '0' && ch <= '3')
986 {
987 row.push_back(ch - '0');
988 }
989 else
990 {
991 row.push_back(-1);
992 }
993 if ((int)row.size() == g.m)
994 break;
995 }
996 if ((int)row.size() != g.m)
997 {
998 throw runtime_error("Row " + to_string(r) + " does not have m entries");
999 }
1000 for (int c = 0; c < g.m; ++c)
1001 {
1002 g.clues[g.cellIndex(r, c)] = row[c];
1003 }
1004 }
1005 return g;
1006}
