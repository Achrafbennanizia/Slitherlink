17 int Solver::calculateOptimalParallelDepth()
18 {
19 int totalCells = grid.n * grid.m;
20 int clueCount = count_if(grid.clues.begin(), grid.clues.end(), [](int c)
21 { return c >= 0; });
22 double density = (double)clueCount / totalCells;
23
24 int depth = (totalCells <= 25) ? 8 : (totalCells <= 49) ? 12
25 : (totalCells <= 64) ? 14
26 : (totalCells <= 100) ? 32
27 : (totalCells <= 144) ? 34
28 : (totalCells <= 225) ? 36
29 : 38;
30
31 if (density < 0.3)
32 depth += 6; // Sparse puzzles need more parallelism
33 return max(10, min(45, depth));
34 }

36 void Solver::buildEdges()
37 {
38 int n = grid.n, m = grid.m;
39 numPoints = (n + 1) * (m + 1);
40 horizEdgeIndex.assign((n + 1) * m, -1);
41 vertEdgeIndex.assign(n * (m + 1), -1);
42 cellEdges.assign(n * m, {});
43 pointEdges.assign(numPoints, {});
44 edges.clear();
45 clueCells.clear();
46 clueCells.reserve(grid.clues.size());
47
48 auto pointId = [m](int r, int c)
49 { return r * (m + 1) + c; };
50 int idx = 0;
51
52 // Build horizontal edges
53 for (int r = 0; r <= n; ++r)
54 for (int c = 0; c < m; ++c)
55 {
56 Edge e{pointId(r, c), pointId(r, c + 1),
57 (r > 0) ? grid.cellIndex(r - 1, c) : -1,
58 (r < n) ? grid.cellIndex(r, c) : -1};
59 edges.push_back(e);
60 horizEdgeIndex[r * m + c] = idx;
61 if (e.cellA >= 0)
62 cellEdges[e.cellA].push_back(idx);
63 if (e.cellB >= 0)
64 cellEdges[e.cellB].push_back(idx);
65 pointEdges[e.u].push_back(idx);
66 pointEdges[e.v].push_back(idx);
67 idx++;
68 }
69
70 // Build vertical edges
71 for (int r = 0; r < n; ++r)
72 for (int c = 0; c <= m; ++c)
73 {
74 Edge e{pointId(r, c), pointId(r + 1, c),
75 (c > 0) ? grid.cellIndex(r, c - 1) : -1,
76 (c < m) ? grid.cellIndex(r, c) : -1};
77 edges.push_back(e);
78 vertEdgeIndex[r * (m + 1) + c] = idx;
79 if (e.cellA >= 0)
80 cellEdges[e.cellA].push_back(idx);
81 if (e.cellB >= 0)
82 cellEdges[e.cellB].push_back(idx);
83 pointEdges[e.u].push_back(idx);
84 pointEdges[e.v].push_back(idx);
85 idx++;
86 }
87
88 for (size_t i = 0; i < grid.clues.size(); ++i)
89 if (grid.clues[i] >= 0)
90 clueCells.push_back((int)i);
91 }

110 bool Solver::applyDecision(State &s, int edgeIdx, int val) const
111 {
112 if (s.edgeState[edgeIdx] == val)
113 return true;
114 if (s.edgeState[edgeIdx] != 0)
115 return false;
116
117 s.edgeState[edgeIdx] = (char)val;
118
119 const Edge &e = edges[edgeIdx];
120
121 s.pointUndecided[e.u]--;
122 s.pointUndecided[e.v]--;
123 if (e.cellA >= 0)
124 s.cellUndecided[e.cellA]--;
125 if (e.cellB >= 0)
126 s.cellUndecided[e.cellB]--;
127
128 if (val == 1)
129 {
130 int du = ++s.pointDegree[e.u];
131 int dv = ++s.pointDegree[e.v];
132 if (du > 2 || dv > 2)
133 return false;
134
135 if (e.cellA >= 0)
136 {
137 int cnt = ++s.cellEdgeCount[e.cellA];
138 if (grid.clues[e.cellA] >= 0 && cnt > grid.clues[e.cellA])
139 return false;
140 }
141 if (e.cellB >= 0)
142 {
143 int cnt = ++s.cellEdgeCount[e.cellB];
144 if (grid.clues[e.cellB] >= 0 && cnt > grid.clues[e.cellB])
145 return false;
146 }
147 }
148 return true;

215 bool Solver::propagateConstraints(State &s) const
216 {
217 for (int cell : clueCells)

302 s.edgeState[eidx] = -1;
303 const Edge &e = edges[eidx];

321 }
322 }
323 }
324 }
325

332 int undecided = s.pointUndecided[ptIdx];
333
334 if (deg >= 2 || (deg == 0 && undecided == 0))
335 continue;
336
337 if (deg == 1 && undecided == 1)
338 {
339 for (int eidx : pointEdges[ptIdx])
340 {
341 if (s.edgeState[eidx] == 0)
342 {
343 if (!applyDecision(s, eidx, 1))
344 return false;
345
346 const Edge &e = edges[eidx];
347 if (e.cellA >= 0 && !cellQueued[e.cellA] && grid.clues[e.cellA] >= 0)
348 {
349 cellQueue.push_back(e.cellA);
350 cellQueued[e.cellA] = true;
351 }
352 if (e.cellB >= 0 && !cellQueued[e.cellB] && grid.clues[e.cellB] >= 0)
353 {
354 cellQueue.push_back(e.cellB);
355 cellQueued[e.cellB] = true;
356 }
357 int otherPt = (e.u == ptIdx) ? e.v : e.u;
358 if (!pointQueued[otherPt])
359 {
360 pointQueue.push_back(otherPt);
361 pointQueued[otherPt] = true;
362 }
363 }
364 }
365 }
366 else if (deg == 2 && undecided > 0)
367 {
368 for (int eidx : pointEdges[ptIdx])
369 {
370 if (s.edgeState[eidx] == 0)
371 {
372 s.edgeState[eidx] = -1;
373 const Edge &e = edges[eidx];
374 s.pointUndecided[e.u]--;
375 s.pointUndecided[e.v]--;
376 if (e.cellA >= 0)
377 s.cellUndecided[e.cellA]--;
378 if (e.cellB >= 0)
379 s.cellUndecided[e.cellB]--;
380
381 int otherPt = (e.u == ptIdx) ? e.v : e.u;
382 if (!pointQueued[otherPt])
383 {
384 pointQueue.push_back(otherPt);
385 pointQueued[otherPt] = true;
386 }
387 }
388 }
389 }
390 }

395 int Solver::selectNextEdge(const State &s) const
396 {
397 int bestEdge = -1, bestScore = -1000;
398
399 auto scoreCell = [&](int cellIdx) -> int
400 {
401 if (cellIdx < 0 || grid.clues[cellIdx] < 0)
402 return 0;
403 int clue = grid.clues[cellIdx], cnt = s.cellEdgeCount[cellIdx], und = s.cellUndecided[cellIdx];
404 if (und == 0)
405 return 0;
406 int need = clue - cnt;
407 return (need == und || need == 0) ? 2000 : (und == 1) ? 1500
408 : (und <= 2) ? 1000
409 : max(0, 100 - abs(need * 2 - und));
410 };
411
412 for (int i = 0; i < (int)edges.size(); ++i)
413 {
414 if (s.edgeState[i] != 0)
415 continue;
416
417 const Edge &e = edges[i];
418 int degU = s.pointDegree[e.u], degV = s.pointDegree[e.v];
419 int undU = s.pointUndecided[e.u], undV = s.pointUndecided[e.v];
420
421 int score = ((degU == 1 || degV == 1) ? 10000 : 0) +
422 ((degU == 0 && undU == 2) || (degV == 0 && undV == 2) ? 5000 : 0) +
423 scoreCell(e.cellA) + scoreCell(e.cellB);
424
425 if (score > bestScore)
426 {
427 bestScore = score;
428 bestEdge = i;
429 if (bestScore >= 10000)
430 return bestEdge;
431 }
432 }
433 return bestEdge >= 0 ? bestEdge : (int)edges.size();
434 }
435 bool Solver::finalCheckAndStore(State &s)
436 {
437#ifdef USE_TBB
438 bool valid = tbb::parallel_reduce(
439 tbb::blocked_range<size_t>(0, clueCells.size()), true,
440 [&](const tbb::blocked_range<size_t> &r, bool v)
441 {
442 for (size_t i = r.begin(); i < r.end() && v; ++i)
443 if (s.cellEdgeCount[clueCells[i]] != grid.clues[clueCells[i]])
444 v = false;
445 return v;
446 },
447 [](bool a, bool b)
448 { return a && b; });
449 if (!valid)
450 return false;
451#else
452 for (int cell : clueCells)
453 if (s.cellEdgeCount[cell] != grid.clues[cell])
454 return false;
455#endif
456
457 vector<vector<int>> adj(numPoints);
458 int start = -1;
459
460#ifdef USE_TBB
461 tbb::parallel_for(tbb::blocked_range<int>(0, numPoints),
462 [&](const tbb::blocked_range<int> &r)
463 {
464 for (int v = r.begin(); v < r.end(); ++v)
465 adj[v].reserve(s.pointDegree[v]);
466 });
467
468 tbb::spin_mutex startMutex;
469 tbb::parallel_for(tbb::blocked_range<size_t>(0, edges.size()),
470 [&](const tbb::blocked_range<size_t> &r)
471 {
472 for (size_t i = r.begin(); i < r.end(); ++i)
473 {
474 if (s.edgeState[i] == 1)
475 {
476 const Edge &e = edges[i];
477 adj[e.u].push_back(e.v);
478 adj[e.v].push_back(e.u);
479 if (start == -1)
480 {
481 tbb::spin_mutex::scoped_lock lock(startMutex);
482 if (start == -1)
483 start = e.u;
484 }
485 }
486 }
487 });
488#else
489 for (int v = 0; v < numPoints; ++v)
490 adj[v].reserve(s.pointDegree[v]);
491 for (size_t i = 0; i < edges.size(); ++i)
492 {
493 if (s.edgeState[i] == 1)
494 {
495 const Edge &e = edges[i];
496 adj[e.u].push_back(e.v);
497 adj[e.v].push_back(e.u);
498 if (start == -1)
499 start = e.u;
500 }
501 }
502#endif
503 if (start == -1)
504 return false;
505
506 int onEdges = 0;
507#ifdef USE_TBB
508 auto result = tbb::parallel_reduce(
509 tbb::blocked_range<int>(0, numPoints),
510 make_pair(true, 0),
511 [&](const tbb::blocked_range<int> &r, pair<bool, int> res)
512 {
513 for (int v = r.begin(); v < r.end() && res.first; ++v)
514 {
515 int deg = adj[v].size();
516 if (deg != 0 && deg != 2)
517 res.first = false;
518 res.second += deg;
519 }
520 return res;
521 },
522 [](pair<bool, int> a, pair<bool, int> b)
523 {
524 return make_pair(a.first && b.first, a.second + b.second);
525 });
526 if (!result.first)
527 return false;
528 onEdges = result.second / 2;
529#else
530 for (int v = 0; v < numPoints; ++v)
531 {
532 int deg = adj[v].size();
533 if (deg != 0 && deg != 2)
534 return false;
535 onEdges += deg;
536 }
537 onEdges /= 2;
538#endif
539 if (onEdges == 0)
540 return false;
541
542 vector<char> vis(numPoints, 0);
543 int visitedEdges = 0;
544 stack<int> st;
545 st.push(start);
546 vis[start] = 1;
547 while (!st.empty())
548 {
549 int v = st.top();
550 st.pop();
551 for (int to : adj[v])
552 {
553 visitedEdges++;
554 if (!vis[to])
555 {
556 vis[to] = 1;
557 st.push(to);
558 }
559 }
560 }
561
562#ifdef USE_TBB
563 bool allVisited = tbb::parallel_reduce(
564 tbb::blocked_range<int>(0, numPoints), true,
565 [&](const tbb::blocked_range<int> &r, bool v)
566 {
567 for (int i = r.begin(); i < r.end() && v; ++i)
568 if (adj[i].size() == 2 && !vis[i])
569 v = false;
570 return v;
571 },
572 [](bool a, bool b)
573 { return a && b; });
574 if (!allVisited || visitedEdges / 2 != onEdges)
575 return false;
576#else
577 for (int v = 0; v < numPoints; ++v)
578 if (adj[v].size() == 2 && !vis[v])
579 return false;
580 if (visitedEdges / 2 != onEdges)
581 return false;
582#endif
583
584 vector<pair<int, int>> cycle;
585 int cols = grid.m + 1;
586 auto coord = [cols](int id)
587 { return make_pair(id / cols, id % cols); };
588
589 int cur = start, prev = -1;
590 do
591 {
592 cycle.push_back(coord(cur));
593 int next = (adj[cur][0] != prev) ? adj[cur][0] : adj[cur][1];
594 prev = cur;
595 cur = next;
596 } while (cur != start);
597 cycle.push_back(coord(start));
598
599 Solution sol;
600 sol.edgeState = s.edgeState;
601 sol.cyclePoints = cycle;
602
603#ifdef USE_TBB
604 int solNum = ++solutionCount;
605 cout << "\n=== Solution " << solNum << " found! ===\n";
606 printSolution(sol);
607 cout << flush;
608
609 tbbSolutions.push_back(sol);
610 if (!findAll)
611 stopAfterFirst.store(true, memory_order_relaxed);
612#else
613 {
614 lock_guard<mutex> lock(solMutex);
615 int solNum = ++solutionCount;
616
617 // Print solution immediately
618 cout << "\n=== Solution " << solNum << " found! ===\n";
619 Solution tempSol = sol;
620 printSolution(tempSol);
621 cout << flush;
622
623 solutions.push_back(std::move(sol));
624 if (!findAll)
625 {
626 stopAfterFirst.store(true, memory_order_relaxed);
627 }
628 }
629#endif
630 return true;
631 }

633 void Solver::search(State s, int depth)
634 {
635 if (!findAll && stopAfterFirst.load(memory_order_relaxed))
636 return;
637
638 if (!quickValidityCheck(s))
639 return;
640
641 if (!propagateConstraints(s))
642 return;
643
644 int edgeIdx = selectNextEdge(s);
645 if (edgeIdx == (int)edges.size())
646 {
647 finalCheckAndStore(s);
648 return;
649 }
650
651 const Edge &edge = edges[edgeIdx];
652 bool canOff = true;
653 bool canOn = true;
654
655 int degU = s.pointDegree[edge.u];
656 int degV = s.pointDegree[edge.v];
657 int undU = s.pointUndecided[edge.u];
658 int undV = s.pointUndecided[edge.v];
659
660 if ((degU == 1 && undU == 1) || (degV == 1 && undV == 1))
661 canOff = false;
662 if (degU >= 2 || degV >= 2)
663 canOn = false;
664
665 State offState;
666 if (canOff)
667 {
668 offState = s;
669 if (!(applyDecision(offState, edgeIdx, -1) && quickValidityCheck(offState) && propagateConstraints(offState)))
670 canOff = false;
671 }
672
673 State onState;
674 if (canOn)
675 {
676 onState = s;
677 if (!(applyDecision(onState, edgeIdx, 1) && quickValidityCheck(onState) && propagateConstraints(onState)))
678 canOn = false;
679 }
680
681 if (!canOn && !canOff)
682 return;
683 if (canOn && !canOff)
684 {
685 search(std::move(onState), depth + 1);
686 return;
687 }
688 if (!canOn && canOff)
689 {
690 search(std::move(offState), depth + 1);
691 return;
692 }
693
694#ifdef USE_TBB
695 if (depth < maxParallelDepth)
696 {
697 tbb::task_group g;
698 g.run([this, off = std::move(offState), depth]()
699 {
700 State local = off;
701 search(std::move(local), depth + 1); });
702 search(std::move(onState), depth + 1);
703 g.wait();
704 }
705 else
706 {
707 search(std::move(offState), depth + 1);
708 if (!findAll && stopAfterFirst.load(memory_order_relaxed))
709 return;
710 search(std::move(onState), depth + 1);
711 }
712#else
713 bool doParallel = (depth < maxParallelDepth &&
714 activeThreads.load(memory_order_relaxed) < maxThreads);
715 if (doParallel)
716 {
717 activeThreads.fetch_add(1, memory_order_relaxed);
718 auto fut = std::async(std::launch::async, [this, off = std::move(offState), depth]() mutable
719 {
720 search(std::move(off), depth + 1);
721 activeThreads.fetch_sub(1, memory_order_relaxed); });
722 search(std::move(onState), depth + 1);
723 fut.get();
724 }
725 else
726 {
727 search(std::move(offState), depth + 1);
728 if (!findAll && stopAfterFirst.load(memory_order_relaxed))
729 return;
730 search(std::move(onState), depth + 1);
731 }
732#endif
733 }

735 void Solver::run(bool allSolutions)
736 {
737 findAll = allSolutions;
738 stopAfterFirst.store(false, memory_order_relaxed);
739 solutionCount.store(0, memory_order_relaxed);
740
741 buildEdges();
742 maxParallelDepth = calculateOptimalParallelDepth();
743
744#ifdef USE_TBB
745 int numThreads = max(1, (int)thread::hardware_concurrency());
746 cout << "Using Intel oneAPI TBB with " << numThreads << " threads (100% CPU)\n";
747 cout << "Dynamic parallel depth: " << maxParallelDepth << " (optimized for "
748 << grid.n << "x" << grid.m << " puzzle)\n";
749 arena = make_unique<tbb::task_arena>(numThreads);
750 tbbSolutions.clear();
751#endif
752
753 cout << "Searching for " << (allSolutions ? "all solutions" : "first solution") << "...\n"
754 << flush;
755
756 State startState = initialState();
757
758#ifdef USE_TBB
759 arena->execute([this, startState]()
760 { search(startState, 0); });
761
762 solutions.clear();
763 for (const auto &sol : tbbSolutions)
764 solutions.push_back(sol);

770 void Solver::printSolution(const Solution &sol) const
771 {
772 int n = grid.n, m = grid.m;
773 auto isHorizOn = [&](int r, int c) -> bool
774 {
775 int idx = horizEdgeIndex[r * m + c];
776 return sol.edgeState[idx] == 1;
777 };
778 auto isVertOn = [&](int r, int c) -> bool
779 {
780 int idx = vertEdgeIndex[r * (m + 1) + c];
781 return sol.edgeState[idx] == 1;
782 };
783
784 for (int r = 0; r <= n; ++r)
785 {
786 string line;
787 for (int c = 0; c < m; ++c)
788 {
789 line += "+";
790 line += (isHorizOn(r, c) ? "-" : " ");
791 }
792 line += "+";
793 cout << line << "\n";
794
795 if (r == n)
796 break;
797
798 string vline;
799 for (int c = 0; c < m; ++c)
800 {
801 vline += (isVertOn(r, c) ? "|" : " ");
802 int clue = grid.clues[grid.cellIndex(r, c)];
803 char ch = ' ';
804 if (clue >= 0)
805 ch = char('0' + clue);
806 vline += ch;
807 }
808 vline += (isVertOn(r, m) ? "|" : " ");
809 cout << vline << "\n";
810 }
811
812 cout << "Cycle (point coordinates row,col):\n";
813 for (size_t i = 0; i < sol.cyclePoints.size(); ++i)
814 {
815 auto [r, c] = sol.cyclePoints[i];
816 cout << "(" << r << "," << c << ")";
817 if (i + 1 < sol.cyclePoints.size())
818 cout << " -> ";
819 }
820 cout << "\n";
821 }
